Conversion en niveaux de gris




let grayImage = null; function openGrayscaleTool() { document.getElementById("grayscaleTool").style.display = "block"; } document.getElementById("grayInput").addEventListener("change", function (e) { const file = e.target.files[0]; const img = new Image(); img.onload = function () { const canvas = document.getElementById("grayCanvas"); const ctx = canvas.getContext("2d"); canvas.width = img.width; canvas.height = img.height; ctx.drawImage(img, 0, 0); grayImage = ctx.getImageData(0, 0, canvas.width, canvas.height); }; img.src = URL.createObjectURL(file); }); function convertToGray() { if (!grayImage) return; const canvas = document.getElementById("grayCanvas"); const ctx = canvas.getContext("2d"); let data = grayImage.data; for (let i = 0; i < data.length; i += 4) { const r = data[i]; const g = data[i + 1]; const b = data[i + 2]; // formule luminance const gray = 0.299 * r + 0.587 * g + 0.114 * b; data[i] = data[i + 1] = data[i + 2] = gray; } ctx.putImageData(grayImage, 0, 0); } function downloadGray() { const canvas = document.getElementById("grayCanvas"); const link = document.createElement("a"); link.download = "image-grayscale.png"; link.href = canvas.toDataURL(); link.click(); }
